CREATE SCHEMA `cuvic` DEFAULT CHARACTER SET utf8;

CREATE TABLE `cuvic`.`user_list` (
  `id` VARCHAR(10) NOT NULL,
  `password` VARCHAR(10) NULL,
  `name` VARCHAR(10) NULL,
  `gender` VARCHAR(2) NULL,
  `phone_num` VARCHAR(11) NULL,
  `email` VARCHAR(100) NULL DEFAULT '-',
  `birth` DATE NULL,
  `club_num` INT(11) NULL,
  `nickname` VARCHAR(10) NULL,
  `work_place` VARCHAR(45) NULL DEFAULT '-',
  `open_state` VARCHAR(45) NULL,
  `img_name` VARCHAR(100) NULL DEFAULT '-',
  `comment` VARCHAR(100) NULL DEFAULT '-',
  PRIMARY KEY (`id`))
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8;


CREATE TABLE `cuvic`.`active_record_list` (
  `title` VARCHAR(200) NOT NULL,
  `date` DATE NULL,
  `member` VARCHAR(200) NULL,
  `detail` VARCHAR(200) NULL,
  PRIMARY KEY (`title`))
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8;

CREATE TABLE `cuvic`.`picture_board` (
  `cnt` INT NOT NULL AUTO_INCREMENT,
  `writer_name` VARCHAR(20) NOT NULL,
  `date` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `title` VARCHAR(100) NOT NULL,
  `contents` VARCHAR(10000) NOT NULL,
  `folder_name` VARCHAR(200) NOT NULL,
  `views` INT NOT NULL DEFAULT 0,
  PRIMARY KEY (`cnt`))
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8;

DEFAULT CHARACTER SET = utf8;

DEFAULT CHARACTER SET = utf8;
