CREATE SCHEMA `cuvic` DEFAULT CHARACTER SET utf8;

CREATE TABLE `cuvic`.`user_list` (
  `id` VARCHAR(10) NOT NULL,
  `password` VARCHAR(10) NOT NULL,
  `name` VARCHAR(10) NOT NULL,
  `gender` VARCHAR(2) NOT NULL,
  `phone_num` VARCHAR(11) NOT NULL DEFAULT '-',
  `email` VARCHAR(100) NOT NULL DEFAULT '-',
  `birth` DATE NULL,
  `club_num` INT(11) NOT NULL DEFAULT '0',
  `nickname` VARCHAR(10) NOT NULL DEFAULT '-',
  `work_place` VARCHAR(45) NOT NULL DEFAULT '-',
  `img_name` VARCHAR(100) NOT NULL DEFAULT '-',
  `comment` VARCHAR(100) NOT NULL DEFAULT '-',
  PRIMARY KEY (`id`))
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8;
